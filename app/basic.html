// key code (official HuggingFace API)

// npm install @huggingface/transformers@4.0.1 // OR import from CDN as shown below import { AutoProcessor, Gemma4ForConditionalGeneration, // Gemma 4 specific class TextStreamer, } from "@huggingface/transformers"; const model_id = "onnx-community/gemma-4-E2B-it-ONNX"; const processor = await AutoProcessor.from_pretrained(model_id); const model = await Gemma4ForConditionalGeneration.from_pretrained( model_id, { dtype: "q4f16", device: "webgpu" } ); const messages = [ { role: "system", content: "You are an IRS expert." }, { role: "user", content: "What is a W-2 form?" }, ]; const inputs = processor.apply_chat_template(messages, { tokenize: true, add_generation_prompt: true, return_dict: true, }); const streamer = new TextStreamer(processor.tokenizer, { skip_prompt: true, skip_special_tokens: true, callback_function: (t) => console.log(t), }); await model.generate({ ...inputs, max_new_tokens: 512, do_sample: true, temperature: 0.7, streamer, });
Requires Chrome 113+ or Edge 113+. Firefox: enable dom.webgpu.enabled. Safari: partial.
